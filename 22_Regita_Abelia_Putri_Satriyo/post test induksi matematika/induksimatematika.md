	1.	p(n) = 2 + 4 + 6 + .... + 2n = n(n + 1)
	
	p(n + 1)    = n(n + 1)  
				= (n + 1)((n + 1) + 1)
				= (n + 1)(n + 2)
				= (n² + n + 2n + 2)
				= (n² + 3n + 2)
				= (n² + n + 2n + 2)
				= (n² + n) + (2n + 2)
				= n(n + 1) + 2(n + 1)
				=   p(n)   + 2(n + 1)
	p(n + 1) 	= 2(n + 1) = 2 + 4 + 6 + ... + 2n + 2(n + 1)

	2. p(n) = 1 + 4 + 9 + ... +  n² = 1/6n (n + 1)(2n + 1)

	p(n + 1) 	= 1/6 n(n  + 1)(2n + 1) 
				= 1/6 (n + 1)((n + 1) + 1)(2(n + 1) + 1)
				= 1/6 (n + 1)(n + 2)((2n + 2) + 1)
				= 1/6 (n + 1)(n + 2)(2n + 3)
				= 1/6 (n + 1)(n² + 4n + 3n + 6)
				= 1/6 (n + 1)(n² + 7n + 6)
				= 1/6 (n + 1)(n² + n + 6n + 6)
				= 1/6 (n + 1)(n² + n) + (6n + 6)
				= 1/6 (n + 1) n(2n + 1) + 6(n + 1)
				=         p(n)          + 6(n + 1)
	p(n + 1)= 1 + 4 + 9 + ... + n² + 6(n + 1)
---

	3. p(n) = 2 + 6 + 12 + ... + n(n + 1) = 1/3n (n + 1)(n + 2)

	p(n + 1) = 1/3 n(n + 1)(n + 2)  
			 = 1/3 (n + 1)((n + 1) + 1)((n + 1) + 2)
			 = 1/3 (n + 1)(n + 2)(n + 3)
			 = 1/3 (n + 1)(n² + 2n + 3n + 6)
			 = 1/3 (n + 1)(n² + 5n + 6)
			 = 1/3 (n + 1)(n² + 2n + 3n + 6)
			 = 1/3 (n + 1)(n² + 2n) + (3n + 6)
			 = 1/3 (n + 1) n(n + 2) + 3(n + 2)
			 =          p(n)        + 3(n + 2)
	p(n + 1) = 2 + 6 + 12 + ... + n(n + 1) + 3(n + 2)
